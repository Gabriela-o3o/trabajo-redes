# Variables
CXX = g++
CXXFLAGS = -Wall -g
CLIENTE_TARGET = cliente
SERVER_TARGET = server
CLIENTE_SOURCES = cliente.cpp
SERVER_SOURCES = server.cpp
CLIENTE_OBJECTS = $(CLIENTE_SOURCES:.cpp=.o)
SERVER_OBJECTS = $(SERVER_SOURCES:.cpp=.o)

# Reglas principales
all: $(CLIENTE_TARGET) $(SERVER_TARGET)

$(CLIENTE_TARGET): $(CLIENTE_OBJECTS)
	$(CXX) $(CXXFLAGS) -o $(CLIENTE_TARGET) $(CLIENTE_OBJECTS)

$(SERVER_TARGET): $(SERVER_OBJECTS)
	$(CXX) $(CXXFLAGS) -o $(SERVER_TARGET) $(SERVER_OBJECTS)
